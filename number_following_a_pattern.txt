//User function Template for Java
class Solution{
    static String printMinNumberForPattern(String s){
        // code here
         Stack<Integer> st=new Stack<>();
         int num=1;
         String ans="";
         for(int i=0;i<s.length();i++){
             char ch=s.charAt(i);
             if(ch=='D'){
                 st.push(num);
                num++;
             }
             else{
                   st.push(num);
                num++;
                
                 while(st.size()>0){
                     ans+=(st.pop());
                 }
                 
             }
         }
         st.push(num);
         while(st.size()>0){
                     ans+=(st.pop());
                 } 
                 return ans;
}
}
